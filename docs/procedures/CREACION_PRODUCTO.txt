2) BUSCAR EL NOMBRE DEL MATERIAL CON LIKE EN LA TABLA MATERIALES

 
 -- LA PREGUNTA 2
 SET SERVEROUTPUT ON;
 CREATE OR REPLACE PROCEDURE MOSTRAR_MATERIALES (rellenar_material VARCHAR2)
 IS
 BEGIN
      SELECT NOMBRE
      FROM MATERIALES
      WHERE NOMBRE LIKE rellenar_material;
 END;
